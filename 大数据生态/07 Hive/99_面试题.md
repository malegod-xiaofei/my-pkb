窗口函数

使用语法 :

- Row_number :
从小到大依次排序 , 忽略相等情况

SELECT mid.num, mid.remark, row_number() OVER (ORDER BY num DESC)

FROM (

SELECT count(1) AS num, remark AS remark

FROM wb_user

WHERE remark != 'null'

GROUP BY remark

) mid;

- Rank
遇到相等情况 , 名次会留下空位

SELECT mid.num, mid.remark, rank() OVER (ORDER BY num DESC)

FROM (

SELECT count(1) AS num, remark AS remark

FROM wb_user

WHERE remark != 'null'

GROUP BY remark

) mid;

- dense_rank
SELECT mid.num, mid.remark, dense_rank() OVER (ORDER BY num DESC)

FROM (

SELECT count(1) AS num, remark AS remark

FROM wb_user

WHERE remark != 'null'

GROUP BY remark

) mid;
