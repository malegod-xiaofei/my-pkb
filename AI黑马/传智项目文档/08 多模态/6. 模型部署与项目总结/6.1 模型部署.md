### TorchServe介绍
+ 关于TorchServe的模块架构, 如下图所示:

![]()

---

+ TorchServe 是一个为PyTorch模型提供服务的灵活且易于使用的工具. TorchServe 模型主要由如下几部分组成:
    - 1 - 前端: TorchServe的请求/响应处理组件。服务组件的这一部分处理来自客户端的请求/响应，并管理模型的生命周期.
    - 2 - 模型推断worker: 这些worker负责执行对模型的实际推断。这些是模型的实际运行实例.
    - 3 - 模型: 模型可以为torchscript model 或者正常导出的模型，这些模型可以听歌自定义的数据预处理和后处理以及任何其他模型构件。模型可以从云存储或者本地主机加载.
    - 4 - 插件: 这些是自定义端点或authz/authn或批处理算法，可以在启动时放入TorchServe.
    - 5 - 模型仓库: 这是一个存放所有可加载模型的目录.

```plain
mport argparse
import os
import sys
import cv2
import numpy as np
import torch
from maskrcnn_benchmark.config import cfg
from maskrcnn_benchmark.layers import nms
from maskrcnn_benchmark.modeling.detector import build_detection_model
from maskrcnn_benchmark.structures.image_list import to_image_list
from maskrcnn_benchmark.utils.model_serialization import load_state_dict
from PIL import Image

class FeatureExtractor:
    MAX_SIZE = 1333
    MIN_SIZE = 800

    # 配置文件以及模型加载
    def _build_detection_model(self, model_file, cfg_file):
        # 加载配置文件
        cfg.merge_from_file(cfg_file)
        cfg.freeze()
        # 加载配置文件
        model = build_detection_model(cfg)
        # 加载模型参数
        checkpoint = torch.load(model_file, map_location=torch.device("cpu"))
        load_state_dict(model, checkpoint.pop("model"))
        model.to("cuda")
        model.eval()
        return model
    # 特征初始化模块
    def __init__(self, model_file, cfg_file):
        self.detection_model = self._build_detection_model( model_file, cfg_file  )

    # 图像格式变化
    def _image_transform(self, path):
        img = Image.open(path)
        im = np.array(img).astype(np.float32)
        if im.shape[-1] > 3:
            im = np.array(img.convert("RGB")).astype(np.float32)
        # IndexError:灰度图像有太多的索引，数组
        if len(im.shape) < 3:
            im = np.repeat(im[:, :, np.newaxis], 3, axis=2)
        # 图像转置
        im = im[:, :, ::-1]
        im -= np.array([102.9801, 115.9465, 122.7717])
        im_shape = im.shape
        im_height = im_shape[0]
        im_width = im_shape[1]
        im_size_min = np.min(im_shape[0:2])
        im_size_max = np.max(im_shape[0:2])
        # 基于最小尺寸的比例
        im_scale = self.MIN_SIZE / im_size_min
        # 阻止的轴大于max_size，如果更大，缩小它
        if np.round(im_scale * im_size_max) > self.MAX_SIZE:
            im_scale = self.MAX_SIZE / im_size_max
        # 对图片进行resize
        im = cv2.resize(
            im, None, None, fx=im_scale, fy=im_scale, interpolation=cv2.INTER_LINEAR
        )
        # 更换维度
        img = torch.from_numpy(im).permute(2, 0, 1)
        # 返回宽和高
        im_info = {"width": im_width, "height": im_height}

        return img, im_scale, im_info

    def _process_feature_extraction(
        self, output, im_scales, im_infos, feature_name="fc6", conf_thresh=0
    ):
        # 获取输出proposals的侯选框
        batch_size = len(output[0]["proposals"])
        # 输出每一张图片的proposals的box长度
        n_boxes_per_image = [len(boxes) for boxes in output[0]["proposals"]]
        # 输出置信度的得分
        score_list = output[0]["scores"].split(n_boxes_per_image)
        # 对置信度缩放到0-1
        score_list = [torch.nn.functional.softmax(x, -1) for x in score_list]
        # 输出每一张图片中n个box的特征embedding
        feats = output[0][feature_name].split(n_boxes_per_image)
        cur_device = score_list[0].device
        feat_list = []
        info_list = []
        for i in range(batch_size):
            # 对输出的候选框除以缩放的尺度
            dets = output[0]["proposals"][i].bbox / im_scales[i]
            # 取到 score_list中每一个score的得分
            scores = score_list[i]
            # max_conf 为最大置信度
            max_conf = torch.zeros(scores.shape[0]).to(cur_device)
            # 创建置信度过滤矩阵
            conf_thresh_tensor = torch.full_like(max_conf, conf_thresh)
            # 开始的index为1
            start_index = 1
            # Column 0 of the scores matrix is for the background class
            # 对于backgroud 的类，得分矩阵的第0列
            # if self.args.background:
            #     start_index = 0
            for cls_ind in range(start_index, scores.shape[1]):
                cls_scores = scores[:, cls_ind]
                # 将IOU 交并比小于0.5的去掉
                keep = nms(dets, cls_scores, 0.5)
                # 将比max_conf大的
                max_conf[keep] = torch.where(
                    # Better than max one till now and minimally greater
                    # than conf_thresh
                    (cls_scores[keep] > max_conf[keep])
                    & (cls_scores[keep] > conf_thresh_tensor[keep]),
                    cls_scores[keep],
                    max_conf[keep],
                )
            # 对类别概率分值和索引进行排序
            sorted_scores, sorted_indices = torch.sort(
                max_conf, descending=True)
            # box的数目
            num_boxes = (sorted_scores[: 100] != 0).sum()
            # 每一个box的索引值
            keep_boxes = sorted_indices[: 100]
            # 特征list添加
            feat_list.append(feats[i][keep_boxes])
            # bbox的得分
            bbox = output[0]["proposals"][i][keep_boxes].bbox / im_scales[i]
            #  使用得分预测类别标签
            objects = torch.argmax(scores[keep_boxes][:, start_index:], dim=1)
            # 得到相应的info_list
            info_list.append(
                {
                    "bbox": bbox.cpu().numpy(),
                    "num_boxes": num_boxes.item(),
                    "objects": objects.cpu().numpy(),
                    "cls_prob": scores[keep_boxes][:, start_index:].cpu().numpy(),
                    "image_width": im_infos[i]["width"],
                    "image_height": im_infos[i]["height"],
                }
            )
        return feat_list, info_list
    def get_detectron_features(self, image_paths):
        img_tensor, im_scales, im_infos = [], [], []
        for image_path in image_paths:
            # 图像tensor大小、图像缩放尺度、图像信息
            im, im_scale, im_info = self._image_transform(image_path)
            img_tensor.append(im)
            im_scales.append(im_scale)
            im_infos.append(im_info)

        current_img_list = to_image_list(img_tensor, size_divisible=32)
        current_img_list = current_img_list.to("cuda")

        with torch.no_grad():
            output = self.detection_model(current_img_list)

        feat_list = self._process_feature_extraction(
            output,
            im_scales,
            im_infos
        )
        return feat_list

    def extract_features(self, image_file):
        # 对待处理图像文件夹中每一个图像进行处理
        features, infos = self.get_detectron_features([image_file])        
        return features, infos
```

import sys
# 整个项目的主目录
sys.path.append("/home/multimodel/")
import time
import numpy as np
from common.utilfuc import get_current_device
import torch
from models.visual_bert import VisualBERT_deploy
from dataset.sample import SampleList, Sample, BatchCollator
from common.configuration import load_yaml
from singlefeature.imageFeatures import FeatureExtractor
from dataset.processors.bert_processors import BertTokenizer


# 对于单条样本构建输入
class singleInput():
    def __init__(self, config):
        # 构建图像特征提取器
        self.imagefeature_extractor = FeatureExtractor(
            config.imagefeatures.model_path, config.imagefeatures.cfg_file)
        # 构建文本特征抽取器
        self.text_processor = BertTokenizer(config.dataset)

    def process_features(self, text, image_file):
        # 基于图像提取特征和特征信息
        features, infos = self.imagefeature_extractor.extract_features(
            image_file)
        item = {}
        item["image_feature_0"] = features[0]
        item["image_info_0"] = infos[0]
        # 处理文本信息
        processed_text = self.text_processor({"text": text})
        # 获取当前的Sample 信息
        current_sample = Sample()
        # 更新current_sample 的text 字段
        current_sample.text = processed_text["text"]
        # 更新bert-tokenizer 处理后的token\input_ids\input_mask\segment_ids\lm_label_ids\tokens
        if "input_ids" in processed_text:
            current_sample.update(processed_text)
        # 更新相应的图像信息
        current_sample.update(item)
        # 更新SampleList.to 设备id
        return SampleList([current_sample]).to(get_current_device())

#### 第3步: 编写VisualBertHandler处理类
+ 代码文件路径: /home/multimodal/deploy/VisualBertHandler.py

```plain
import sys
sys.path.append("/home/multimodel/")
from singlefeature.singleMultiInput import *
from common.configuration import load_yaml
from models.visual_bert import VisualBERT_deploy
from abc import ABC
import json
import logging
import os
import ast
import torch
from ts.torch_handler.base_handler import BaseHandler
from captum.attr import LayerIntegratedGradients
logger = logging.getLogger(__name__)

class VisualBertHandler(BaseHandler, ABC):
    def __init__(self):
        super(VisualBertHandler, self).__init__()
        self.initialized = False

    def initialize(self, ctx):
        # 设置manifest 配置文件
        self.manifest = ctx.manifest
        properties = ctx.system_properties
        # 设置model_dir以及模型文件
        model_dir = properties.get("model_dir")
        serialized_file = self.manifest["model"]["serializedFile"]
        model_pt_path = os.path.join(model_dir, serialized_file)
        # 设置setup_config.json 配置文件
        setup_config_path = os.path.join(model_dir, "setup_config.json")
        if os.path.isfile(setup_config_path):
            with open(setup_config_path) as setup_config_file:
                self.setup_config = json.load(setup_config_file)
        else:
            logger.warning("Missing the setup_config.json file.")

        # 设置设备id为GPU 还是CPU    
        self.device = torch.device(
            "cuda:" + str(self.setup_config["gpu_id"])  #properties.get("gpu_id"))
            if torch.cuda.is_available() and self.setup_config["gpu_id"]#properties.get("gpu_id")
            else "cpu"
        )
        # 加载模型配置文件
        visual_bert_cfg = load_yaml(
            os.path.join(model_dir, "visual_bert.yaml"))
        # 根据配置文件中的相应配置设置模型加载的方式
        # 如果为torchscript 模型文件
        if self.setup_config["save_mode"] == "torchscript":
            self.model = torch.jit.load(model_pt_path)
        # 如果为正常模型
        elif self.setup_config["save_mode"] == "pretrained":
            if self.setup_config["mode"] == "visualbert":
                # 设置模型预训练
                self.model = VisualBERT_deploy.from_pretrained(
                    model_pt_path, visual_bert_cfg)  
            else:
                logger.warning("Missing the operation mode.")
        else:
            logger.warning("Missing the checkpoint or state_dict.")

        # 设置输入Input的启动进程
        self.multi_process = singleInput(visual_bert_cfg)
        # 设置模型为eval
        self.model.to(self.device)
        self.model.eval()
        logger.info(
            "Transformer model from path %s loaded successfully", model_dir
        )
        # 读取返回结果的映射文件
        mapping_file_path = os.path.join(model_dir, "index_to_name.json")
        if os.path.isfile(mapping_file_path):
            with open(mapping_file_path) as f:
                self.mapping = json.load(f)
        else:
            logger.warning("Missing the index_to_name.json file.")
        self.initialized = True

    def preprocess(self, requests):
        input_batch = None
        # 遍历requests 请求
        for idx, data in enumerate(requests):
            # 获取text 参数
            input_text = data.get("text")
            # 获取image 参数
            input_image = data.get("image")

            if isinstance(input_text, (bytes, bytearray)):
                input_text = input_text.decode('utf-8')

            if isinstance(input_image, (bytes, bytearray)):
                input_image = input_image.decode('utf-8')

            # 基于单条样本构建输入请求
            input_batch = self.multi_process.process_features(input_text, input_image)

        return input_batch

    def inference(self, input_batch):
        inferences = []

        # 处理visualbert的推断请求
        if self.setup_config["mode"] == "visualbert":
            prediction = self.model(input_batch)
            prediction = prediction['scores'].detach().cpu().numpy()

            index_predict = np.argmax(prediction, axis=1)
            index_predict = str(index_predict[0])
            inferences.append(self.mapping[index_predict])

        return inferences
```
