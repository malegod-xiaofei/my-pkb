```plain
class BertSelfAttentionJit(BertSelfAttention):
    # 更改张量的形状
    def transpose_for_scores(self, x: Tensor) -> Tensor:
        new_x_shape = x.size()[:-1] + (
            self.num_attention_heads,
            self.attention_head_size,
        )
        x = x.view(new_x_shape)
        return x.permute(0, 2, 1, 3)

    def forward(
        self,
        hidden_states: Tensor,
        attention_mask: Optional[Tensor] = None,
        head_mask: Optional[Tensor] = None,
        encoder_hidden_states: Optional[Tensor] = None,
        encoder_attention_mask: Optional[Tensor] = None,
    ) -> Tuple[Tensor, Tensor]:

        # 按照注意力计算公式进行Q, K, V的相关计算
        mixed_query_layer = self.query(hidden_states)
        # 此处采用 cross-attention 的模式, K和V来自Encoder.
        # attention mask需要的是Encoder中的padding mask
        if encoder_hidden_states is not None:
            mixed_key_layer = self.key(encoder_hidden_states)
            mixed_value_layer = self.value(encoder_hidden_states)
            attention_mask = encoder_attention_mask
        else:
            mixed_key_layer = self.key(hidden_states)
            mixed_value_layer = self.value(hidden_states)

        query_layer = self.transpose_for_scores(mixed_query_layer)
        key_layer = self.transpose_for_scores(mixed_key_layer)
        value_layer = self.transpose_for_scores(mixed_value_layer)

        # 按照经典计算公式得出注意力分布的分数
        attention_scores = torch.matmul(
            query_layer, key_layer.transpose(-1, -2))
        attention_scores = attention_scores / \
            math.sqrt(self.attention_head_size)
        if attention_mask is not None:
            # 这一步操作为padding mask, 其主要作用为在进行前向运算时去除
            # 字符当中含有0的数据的影响，由于图像没有空的，所以对图像无用
            attention_scores = attention_scores + attention_mask

        # print("attention_scores out ",attention_scores, attention_scores.shape)
        attention_probs = nn.functional.softmax(attention_scores, dim=-1)

        attention_probs = self.dropout(attention_probs)

        if head_mask is not None:
            attention_probs = attention_probs * head_mask

        # 完成注意力公式中最后的乘以V的操作
        context_layer = torch.matmul(attention_probs, value_layer)

        # 先执行.permute(), 或者.transpose(), 需要中间加一个.contiguous(), 后面才能执行.view()
        context_layer = context_layer.permute(0, 2, 1, 3).contiguous()
        new_context_layer_shape = context_layer.size()[
            :-2] + (self.all_head_size,)
        context_layer = context_layer.view(new_context_layer_shape)

        outputs = (context_layer, attention_probs)
        return outputs
```

```plain
class BertAttentionJit(BertAttention):
    def __init__(self, config):
        super().__init__(config)
        # 设置attention主层, 为了适应核心SelfAttention运算而采用的计算方案
        self.self = BertSelfAttentionJit(config)
        # 设置线性, ln, dropout, 实现short-connect
        self.output = BertSelfOutput(config)
        self.pruned_heads = set()

    def forward(
        self,
        hidden_states: Tensor,
        attention_mask: Optional[Tensor] = None,
        head_mask: Optional[Tensor] = None,
        encoder_hidden_states: Optional[Tensor] = None,
        encoder_attention_mask: Optional[Tensor] = None,
    ) -> List[Tensor]:
        # 首先计算自注意力层的输出张量
        self_outputs = self.self(
            hidden_states,
            attention_mask,
            head_mask,
            encoder_hidden_states,
            encoder_attention_mask,
        )
        # 第0个位置的张量经过融合隐藏层输出得到注意力输出结果
        attention_output = self.output(self_outputs[0], hidden_states)
        # 注意力输出结果和除去第0个位置之后的所有张量进行融合, 得到最终的输出张量.
        outputs = (attention_output,) + self_outputs[1:]  
        return outputs
```

```plain
class BertLayerJit(BertLayer):
    def __init__(self, config):
        super().__init__(config)
        # 设置Bert的Attention的Jit 层
        self.attention = BertAttentionJit(config)
        # 设置是否使用decoder 层
        self.is_decoder = config.is_decoder
        if self.is_decoder:
            self.crossattention = BertAttentionJit(config)

    def forward(
        self,
        hidden_states: Tensor,
        attention_mask: Optional[Tensor] = None,
        head_mask: Optional[Tensor] = None,
        encoder_hidden_states: Optional[Tensor] = None,
        encoder_attention_mask: Optional[Tensor] = None,
    ) -> List[Tensor]:

        self_attention_outputs = self.attention(hidden_states, attention_mask, head_mask)

        attention_output = self_attention_outputs[0]
        outputs = self_attention_outputs[1:]

        # 相当于feed-forward 层
        intermediate_output = self.intermediate(attention_output)
        layer_output = self.output(intermediate_output, attention_output)
        outputs = (layer_output,) + outputs
        return outputs
```

```plain
# Bertencoder 的jit版本
class BertEncoderJit(BertEncoder):
    def __init__(self, config):
        super().__init__(config)
        # 设置输出的attentions
        self.output_attentions = config.output_attentions
        # 设置输出隐藏层
        self.output_hidden_states = config.output_hidden_states
        # 设置输出layer层
        self.layer = nn.ModuleList(
                        [BertLayerJit(config) for _ in range(config.num_hidden_layers)]
                        )

    def forward(
        self,
        hidden_states: Tensor,
        attention_mask: Optional[Tensor],
        encoder_hidden_states: Optional[Tensor] = None,
        encoder_attention_mask: Optional[Tensor] = None,
        output_attentions: bool = False,
        output_hidden_states: bool = False,
        return_dict: bool = False,
        head_mask: Optional[Tensor] = None,
    ) -> Tuple[Tensor]:
        # 初始化隐藏层和注意力张量
        all_hidden_states = ()
        all_attentions = ()
        # 依次计算所有的Encoder Block层
        for i, layer_module in enumerate(self.layer):
            if not torch.jit.is_scripting() and output_hidden_states:
                all_hidden_states = all_hidden_states + (hidden_states,)

            # 每一层都得到一个输出
            layer_outputs = layer_module(
                hidden_states,
                attention_mask,
                None,
                encoder_hidden_states,
                encoder_attention_mask,
            )

            # 仅仅将第一个张量作为当前层的隐藏层输出张量
            hidden_states = layer_outputs[0]

            if not torch.jit.is_scripting() and output_attentions:
                all_attentions = all_attentions + (layer_outputs[1],)

        # 增加最后一层
        if not torch.jit.is_scripting() and output_hidden_states:
            all_hidden_states = all_hidden_states + (hidden_states,)

        outputs = (hidden_states,)
        if not torch.jit.is_scripting():
            if output_hidden_states:
                outputs = outputs + (all_hidden_states,)
            if output_attentions:
                outputs = outputs + (all_attentions,)
        # hidden state, (all hidden states), (all attentions)
        # 输出最后一层隐藏状态, (所有隐藏状态), (所有关注)
        return outputs
```

```plain
class BertEmbeddingsJit(BertEmbeddings):
    def forward(
        self,
        input_ids: Tensor,
        token_type_ids: Optional[Tensor] = None,
        position_ids: Optional[Tensor] = None,
        inputs_embeds: Optional[Tensor] = None,
    ):
        # 得到输入语句张量的维度信息
        if input_ids is not None:
            input_shape = input_ids.size()
        else:
            input_shape = inputs_embeds.size()[:-1]

        # 一般来说都是[batch_size, sequence_length, embedding_dim]
        seq_length = input_shape[1]
        device = inputs_embeds.device if inputs_embeds is not None else input_ids.device

        # 如果没有positional embedding信息, 给一个默认的位置编码张量
        if position_ids is None:
            position_ids = torch.arange(
                seq_length, dtype=torch.long, device=device)
            position_ids = position_ids.unsqueeze(0).expand(input_shape)

        # 如果没有segmentation embedding信息, 给一个默认的语句段编码张量
        if token_type_ids is None:
            token_type_ids = torch.zeros(
                input_shape, dtype=torch.long, device=device)

        # 如果没有token embedding信息, 则经历词嵌入函数进行处理
        if inputs_embeds is None:
            inputs_embeds = self.word_embeddings(input_ids)
        position_embeddings = self.position_embeddings(position_ids)
        token_type_embeddings = self.token_type_embeddings(token_type_ids)

        # 最终BERT的embedding由3个子层编码的加和来得到
        embeddings = inputs_embeds + position_embeddings + token_type_embeddings
        embeddings = self.LayerNorm(embeddings)
        embeddings = self.dropout(embeddings)
        return embeddings
```

```plain
class BertModelJit(BertModel):
    __jit_unused_properties__ = [
        "base_model", "dummy_inputs", "device", "dtype"]

    def __init__(self, config):
        super().__init__(config)
        self.config = config
        self.embeddings = BertEmbeddingsJit(config)
        self.encoder = BertEncoderJit(config)
        self.pooler = BertPooler(config)

        self.init_weights()

    def forward(
        self,
        input_ids: Tensor,
        attention_mask: Optional[Tensor] = None,
        token_type_ids: Optional[Tensor] = None,
        position_ids: Optional[Tensor] = None,
        head_mask: Optional[Tensor] = None,
        inputs_embeds: Optional[Tensor] = None,
        encoder_hidden_states: Optional[Tensor] = None,
        encoder_attention_mask: Optional[Tensor] = None,
    ) -> Tuple[Tensor, Tensor, List[Tensor]]:
        # 进行若干特殊情况的判断和预处理
        if input_ids is not None and inputs_embeds is not None:
            raise ValueError(
                "张量input_ids和inputs_embeds不能同时实例化赋值."
            )
        elif input_ids is not None:
            input_shape = input_ids.size()
        elif inputs_embeds is not None:
            input_shape = inputs_embeds.size()[:-1]
        else:
            raise ValueError(
                "张量input_ids或inputs_embeds至少要有一个完成实例化赋值.")

        # 指定模型的运行设备device
        device = inputs_embeds.device if inputs_embeds is not None else input_ids.device
        if attention_mask is None:
            attention_mask = torch.ones(input_shape, device=device)
        if token_type_ids is None:
            token_type_ids = torch.zeros(
                input_shape, dtype=torch.long, device=device)

        # 关于 attention mask 的维度设定
        # [batch_size, from_seq_length, to_seq_length]
        if attention_mask.dim() == 3:
            extended_attention_mask = attention_mask[:, None, :, :]
        elif attention_mask.dim() == 2:
            # 如果是 padding mask 则保持维度等于 [batch_size, seq_length]
            # 如果当前模型是 Decoder, 则在padding mask后再加上look-ahead mask
            # 如果当前模型是 Encoder, 则通过 broadcast 机制使得mask张量的形状是:
            # [batch_size, num_heads, seq_length, seq_length]
            extended_attention_mask = attention_mask[:, None, None, :]
        else:
            raise ValueError(
                f"错误的张量形状 input_ids (shape {input_shape}) or "
                + f"attention_mask (shape {attention_mask.shape})"
            )

        if not torch.jit.is_scripting():
            extended_attention_mask = extended_attention_mask.to(
                dtype=next(self.parameters()).dtype
            )
        extended_attention_mask = (1.0 - extended_attention_mask) * -10000.0

        # 如果是 2D或3D attention mask, 我们需要利用broadcast机制得到张量形状如下:
        # [batch_size, num_heads, seq_length, seq_length]
        encoder_extended_attention_mask = None

        # 首先经历embedding的词嵌入处理
        embedding_output = self.embeddings(
            input_ids=input_ids,
            position_ids=position_ids,
            token_type_ids=token_type_ids,
            inputs_embeds=inputs_embeds,
        )
        # 然后通过全部的Encoder处理
        encoder_outputs = self.encoder(
            embedding_output,
            attention_mask=extended_attention_mask,
            encoder_hidden_states=encoder_hidden_states,
            encoder_attention_mask=encoder_extended_attention_mask,
        )
        # 最后将Encoder的输出张量的第一个Tensor作为所谓的[CLS]输出, 进行Pooler的处理
        sequence_output = encoder_outputs[0]
        pooled_output = self.pooler(sequence_output)

        # 最终返回的输出张量包含几个部分:
        # sequence_output, pooled_output, (hidden_states), (attentions)
        outputs = (sequence_output, pooled_output, encoder_outputs[1:])
        return outputs
```

```plain
# 将BERT模型替换成Jit版本
def replace_with_jit():
    # 替换方法就是将所有的前向计算函数forward进行替换
    BertEmbeddings.forward = BertEmbeddingsJit.forward
    BertEncoder.forward = BertEncoderJit.forward
    BertLayer.forward = BertLayerJit.forward
    BertAttention.forward = BertAttentionJit.forward
    BertSelfAttention.forward = BertSelfAttentionJit.forward
    BertSelfAttention.transpose_for_scores = BertSelfAttentionJit.transpose_for_scores
    BertModel.forward = BertModelJit.forward
    PreTrainedModel.__jit_unused_properties__ = [
        "base_model",
        "dummy_inputs",
        "device",
        "dtype",
    ]
```

```plain
    if len(tensor.size()) == 2:
        return tensor
    else:
        assert len(tensor.size()) == 3
```

```plain
# 基于BERT的视觉语言嵌入类
class BertVisioLinguisticEmbeddings(BertEmbeddings):
    def __init__(self, config, *args, **kwargs):
        super().__init__(config)
        # 设置type_vocab_size 为2 表示是视觉单元, hidden_size为768 表示隐层
        self.token_type_embeddings_visual = nn.Embedding(config.type_vocab_size, config.hidden_size)

        # 设置position embeddings
        self.position_embeddings_visual = nn.Embedding(config.max_position_embeddings, config.hidden_size)

        # 设置线性层将2048维的映射为768维
        self.projection = nn.Linear(config.visual_embedding_dim, config.hidden_size)

    # 将文本模态的token_type和位置的参数赋值给视觉
    def initialize_visual_from_pretrained(self):
        self.token_type_embeddings_visual.weight = nn.Parameter(
                               deepcopy(self.token_type_embeddings.weight.data), requires_grad=True)

        self.position_embeddings_visual.weight = nn.Parameter(
                               deepcopy(self.position_embeddings.weight.data), requires_grad=True)

    # 对文本模态进行Embedding
    def encode_text(
        self, input_ids: Tensor, token_type_ids: Optional[Tensor] = None) -> Tensor:
        # 输入文本的长度
        seq_length = input_ids.size(1)
        # 产生从1到序列长度个的位置信息
        position_ids = torch.arange(seq_length, dtype=torch.long, device=input_ids.device)
        # 对位置信息进行升维, 并扩展到与input_ids一样的维度
        position_ids = position_ids.unsqueeze(0).expand_as(input_ids)

        # 如果token_type_ids为空，初始化
        if token_type_ids is None:
            token_type_ids = torch.zeros_like(input_ids)

        # 对input_ids embedding化, 默认为调用nn.Embedding(config.vocab_size, config.hidden_size, padding_idx=config.pad_token_id)
        words_embeddings = self.word_embeddings(input_ids)

        # embedding位置, nn.Embedding(config.max_position_embeddings, config.hidden_size)
        position_embeddings = self.position_embeddings(position_ids)

        # embedding Token_type的信息, nn.Embedding(config.type_vocab_size, config.hidden_size)
        token_type_embeddings = self.token_type_embeddings(token_type_ids)

        # 几个embedding矩阵进行相加
        embeddings = words_embeddings + position_embeddings + token_type_embeddings
        return embeddings

    # 对视觉模态进行Embedding
    def encode_image(
        self,
        visual_embeddings: Tensor,
        visual_embeddings_type: Tensor,
        image_text_alignment: Optional[Tensor] = None,) -> Tensor:
        # 将2048 维的图像embedding 映射成768维
        visual_embeddings = self.projection(visual_embeddings)
        # 视觉token_type信息进行压缩
        token_type_embeddings_visual = self.token_type_embeddings_visual(visual_embeddings_type)

        # 获取位置的embedding信息, 这个依赖于图文是否对齐
        position_embeddings_visual = self.get_position_embeddings_visual(
                                     visual_embeddings, image_text_alignment=image_text_alignment)

        # 计算视觉embedding 的相加后信息
        v_embeddings = (visual_embeddings + 
                        position_embeddings_visual + 
                        token_type_embeddings_visual
                       )
        return v_embeddings

    # 该函数主要为视觉文本对齐的embedding信息
    def get_position_embeddings_visual(
        self, visual_embeddings: Tensor, image_text_alignment: Optional[Tensor] = None) -> Tensor:
        # 如果需要视觉文本信息对齐
        if image_text_alignment is not None:
            # image_text_alignment = Batch x image_length x alignment_number.
            # 每一个元素标记了word对应于image特征的位置信息, 在这里-1代表padding值
            image_text_alignment_mask = (
                (image_text_alignment != -1).long().to(image_text_alignment.device)
            )
            # 进行掩码张量的计算
            image_text_alignment = image_text_alignment_mask * image_text_alignment

            # 视觉信息的位置编码张量
            # 张量形状为[batch_size, image_length, alignment length, dim]
            position_embeddings_visual = self.position_embeddings(image_text_alignment) * image_text_alignment_mask.unsqueeze(-1)

            # 在alignment_number的维度上求平均值
            position_embeddings_visual = position_embeddings_visual.sum(2)

            # 在alignment_number的维度上求平均值
            image_text_alignment_mask = image_text_alignment_mask.sum(2)
            image_text_alignment_mask[image_text_alignment_mask == 0] = torch.tensor([1], dtype=torch.long)

            # 防止除零操作
            position_embeddings_visual = (position_embeddings_visual / image_text_alignment_mask.unsqueeze(-1))

            position_ids_visual = torch.zeros(visual_embeddings.size()[:-1],
                                              dtype=torch.long,
                                              device=visual_embeddings.device)

            position_embeddings_visual = (position_embeddings_visual + self.position_embeddings_visual(position_ids_visual))
        # 否则不需要视觉文本信息对齐
        else:
            position_ids_visual = torch.zeros(visual_embeddings.size()[:-1],
                                              dtype=torch.long,
                                              device=visual_embeddings.device)

            position_embeddings_visual = self.position_embeddings_visual(position_ids_visual)

        return position_embeddings_visual

    def forward(
        self,
        input_ids: Tensor,
        token_type_ids: Optional[Tensor] = None,
        visual_embeddings: Optional[Tensor] = None,
        visual_embeddings_type: Optional[Tensor] = None,
        image_text_alignment: Optional[Tensor] = None,) -> Tensor:
        # 首先明确若干参数张量的形状
        # input_ids = [batch_size, sequence_length]
        # token_type_ids = [batch_size, sequence_length]
        # visual_embedding = [batch_size, image_feature_length, image_feature_dim]
        # image_text_alignment = [batch_size, image_feature_length, alignment_dim]

        # 对文本embedding化
        text_embeddings = self.encode_text(input_ids, token_type_ids=token_type_ids)

        # 对视觉信息embedding化, 几个参数注意visual_embeddings_type为视觉的类型信息,
        # image_text_alignment为图文对齐信息
        if visual_embeddings is not None and visual_embeddings_type is not None:
            v_embeddings = self.encode_image(
                visual_embeddings,
                visual_embeddings_type=visual_embeddings_type,
                image_text_alignment=image_text_alignment,
            )

            # 将视觉和图像的embedding 信息进行拼接
            embeddings = torch.cat((text_embeddings, v_embeddings), dim=1)
        # 否则只返回文本的embedding
        else:
            embeddings = text_embeddings
        # 依然采用Transformer中的LayerNorm正则化
        embeddings = self.LayerNorm(embeddings)
        embeddings = self.dropout(embeddings)
        return embeddings
```

```plain
# VisualBERTBase 继承了BertPreTrainedModel, BertPreTrainedModel为Huggingface里一个处理权值初始化以及预训练模型加载的接口
class VisualBERTBase(BertPreTrainedModel):
    def __init__(
        self,
        config,
        visual_embedding_dim=512,
        embedding_strategy="plain",
        bypass_transformer=False,
        output_attentions=False,
        output_hidden_states=False,
    ):
        super().__init__(config)
        self.config = config
        # 视觉embedding维度赋值, 参数为2048
        config.visual_embedding_dim = visual_embedding_dim
        # 参数维度策略为plain
        config.embedding_strategy = embedding_strategy
        # 设置bypass_transformer 为false
        config.bypass_transformer = bypass_transformer
        # 设置输出attention为false
        config.output_attentions = output_attentions
        # 输出的隐藏状态为false
        config.output_hidden_states = output_hidden_states
        # 视觉和文本输入的embedding化
        self.embeddings = BertVisioLinguisticEmbeddings(config)
        # 设置encoer其为BertEncoder的Jit版本, 并对其进行了封装了
        self.encoder = BertEncoderJit(config)
        # 设置pooling层, BertPooler的操作为取出每一个句子的第一个表示做为整个句子的表示
        self.pooler = BertPooler(config)
        # 设置bypass-transformer, 设置之后不利用transformer层
        self.bypass_transformer = config.bypass_transformer
        # 把output_attentions重新设置
        self.output_attentions = self.config.output_attentions
        # 把hidden_state  重新再设置一下
        self.output_hidden_states = self.config.output_hidden_states
        self.init_weights()

    def forward(self, input_ids: Tensor,
                attention_mask: Optional[Tensor] = None,
                token_type_ids: Optional[Tensor] = None,
                visual_embeddings: Optional[Tensor] = None,
                visual_embeddings_type: Optional[Tensor] = None,
                image_text_alignment: Optional[Tensor] = None,
                ) -> Tuple[Tensor, Tensor, List[Tensor]]:
        if attention_mask is None:
            attention_mask = torch.ones_like(input_ids)
        if token_type_ids is None:
            token_type_ids = torch.zeros_like(input_ids)
        # 我们从二维的掩码张量创建一个4维的attention mask 矩阵,
        # 其shape = [batch_size, 1, 1, to_seq_length]
        extended_attention_mask = attention_mask.unsqueeze(1).unsqueeze(2)
        extended_attention_mask = (1.0 - extended_attention_mask) * -10000.0

        embedding_output = self.embeddings(
            input_ids,
            token_type_ids,
            visual_embeddings=visual_embeddings,
            visual_embeddings_type=visual_embeddings_type,
            image_text_alignment=image_text_alignment,
        )

        # 对输入的embedding进行encoder操作
        encoded_layers = self.encoder(
            embedding_output, extended_attention_mask)
        # encoder最后一层输出
        sequence_output = encoded_layers[0]
        # 取出每一句的第一个单词, 做线性映射和激活. 得到的输出可以用来分类等下游任务(即将每个句子的第一个单词的表示作为整个句子的表示)
        pooled_output = self.pooler(sequence_output)
        attn_data_list: List[Tensor] = []
        # 返回sequence_output, pooled_output以及attn_data_list
        return sequence_output, pooled_output, attn_data_list
```

```plain
        self.dropout = nn.Dropout(self.bert.config.hidden_dropout_prob)

        # 分类层
        logits = self.classifier(pooled_output)
        # 输出转化
        reshaped_logits = logits.contiguous().view(-1, self.num_labels)
        # 得到output_dict
        output_dict["scores"] = reshaped_logits
        return output_dict
```

```plain
class VisualBERT(nn.Module):
    # 初始化函数的操作
    def __init__(self, config, model_type='VisualBERT'):
        super().__init__()
        self.config = config
        self.training_head_type = self.config.training_head_type
        self.build()

    def build(self):
        # 模型下游任务微调部分
        self.model = VisualBERTForClassification(self.config.model_config.visual_bert)
        # 初始化模型参数
        self.model.bert.embeddings.initialize_visual_from_pretrained()

    # 加载预训练模型
    @classmethod
    def from_pretrained(cls, checkpoint, config):  # *args, **kwargs):
        cls = VisualBERT
        instance = cls(config)
        instance.is_pretrained = True
        instance.build()

        with PathManager.open(checkpoint, "rb") as f:
            checkpoint = torch.load(f, map_location=lambda storage, loc: storage)

        instance.load_state_dict(checkpoint, strict=False)
        instance.eval()

        return instance

    # 将张量本身做flatten操作, 或者对dim做flatten操作
    def flatten(
        self,
        sample_list: Dict[str, Tensor],
        to_be_flattened: List[str],
        to_be_flattened_dim: List[str],
        ) -> Dict[str, Tensor]:
        for key in to_be_flattened:
            # 要确保这些keys已经存在并设置好, 否则置为None
            sample_list[key] = transform_to_batch_sequence(sample_list[key])

        for key in to_be_flattened_dim:
            # 同上, 对key做dim的flatten操作
            sample_list[key] = transform_to_batch_sequence_dim(sample_list[key])

        return sample_list

    # 添加后向的flatten参数
    def add_post_flatten_params(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        # 图像的mask张量形状, 用来初始化视觉嵌入部分的类型
        sample_list["visual_embeddings_type"] = torch.zeros_like(sample_list["image_mask"])
        # 将文本mask张量, 和图像mask张量做拼接, 生成注意力mask
        attention_mask = torch.cat(
            (sample_list["input_mask"], sample_list["image_mask"]), dim=-1)

        sample_list["attention_mask"] = attention_mask
        if self.training_head_type == "pretraining":
            assert sample_list["masked_lm_labels"].size(-1) == sample_list["input_mask"].size(-1)

            new_lm_labels = torch.ones_like(attention_mask) * -1
            size_masked_lm_labels = sample_list["masked_lm_labels"].size()

            assert len(size_masked_lm_labels) == 2

            new_lm_labels[:size_masked_lm_labels[0], :size_masked_lm_labels[1]] = sample_list["masked_lm_labels"]
            sample_list["masked_lm_labels"] = new_lm_labels
        return sample_list

    # 对BERT模型的相关张量做flatten操作
    def flatten_for_bert(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        to_be_flattened = ["input_ids", "token_type_ids", "input_mask", "image_mask"]

        to_be_flattened_dim = ["visual_embeddings"]

        if self.training_head_type == "pretraining":
            to_be_flattened.append("masked_lm_labels")

        # 把所有数据转化成形状等于[batch_size, sequence_length, dim].
        flattened = self.flatten(sample_list, to_be_flattened, to_be_flattened_dim)
        return flattened

    # 根据BERT的产出张量更新样本值
    def update_sample_list_based_on_head(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        bert_input_ids = sample_list["input_ids"]
        bert_input_mask = sample_list["input_mask"]
        bert_input_type_ids = sample_list["segment_ids"]
        if not torch.jit.is_scripting():
            image_info = getattr(sample_list, "image_info_0", {})
            image_dim_variable = getattr(image_info, "max_features", None)
            image_feat_variable = getattr(sample_list, "image_feature_0", None)
        else:
            image_feat_variable = sample_list["image_feature_0"]
            image_dim_variable = None

        if image_dim_variable is None:
            image_dim_variable = sample_list["image_feature_0"].new_full(
                size=(image_feat_variable.size(0), 1),
                fill_value=image_feat_variable.size(1),
            )
        sample_list["visual_embeddings"] = image_feat_variable
        sample_list["image_dim"] = image_dim_variable
        sample_list["input_ids"] = bert_input_ids
        sample_list["input_mask"] = bert_input_mask
        sample_list["token_type_ids"] = bert_input_type_ids
        return sample_list

    def add_custom_params(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        visual_embeddings = sample_list["visual_embeddings"]
        image_dim = sample_list["image_dim"]
        # 如果是预训练模型, 则增加masked_lm_labels
        if self.training_head_type == "pretraining":
            sample_list["masked_lm_labels"] = sample_list["lm_label_ids"]
        # 添加image mask
        image_mask = torch.arange(
            visual_embeddings.size(-2), device=visual_embeddings.device
            ).expand(visual_embeddings.size()[:-1])

        # 如果图像的维度比image_mask的维度小则进行维度扩展
        if len(image_dim.size()) < len(image_mask.size()):
            image_dim = image_dim.unsqueeze(-1)
            assert len(image_dim.size()) == len(image_mask.size())

        image_mask = image_mask < image_dim
        sample_list["image_mask"] = image_mask.long()

        return sample_list

    # 为了和原始的VisualBERT实现后向兼容.
    @classmethod
    def format_state_key(cls, key):
        return (
            key.replace("bert.bert", "model.bert")
            .replace("bert.cls", "model.cls")
            .replace("bert.classifier", "model.classifier")
        )

    @classmethod
    def format_state_keycopy(cls, key):
        return (
            key.replace("model.model", "model")
            # .replace("bert.cls", "model.cls")
            # .replace("bert.classifier", "model.classifier")
        )

    # 获取优化器参数
    def get_optimizer_parameters(self, config):
        return get_optimizer_parameters_for_bert(self.model, config)

    def forward(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        # import pickle
        # pickle.dump(sample_list, open('sample_01.ckpt','wb'))
        # 更新sample_list名称为新的sample_list名称及处理sample_list中的数据.
        # 这一步操作主要为预处理各种各样产生的字段名称不同, 所以需要统一
        sample_list = self.update_sample_list_based_on_head(sample_list)
        # 为sample_list 增加自定义参数
        sample_list = self.add_custom_params(sample_list)

        # 将"input_ids", "token_type_ids", "input_mask", "image_mask"转化为[batch, sequence_length, dim]
        sample_list = self.flatten_for_bert(sample_list)
        sample_list = self.add_post_flatten_params(sample_list)
        # pickle.dump(sample_list, open('sample_02.ckpt','wb'))

        output_dict = self.model(
            sample_list["input_ids"],
            sample_list["input_mask"],
            sample_list["attention_mask"],
            sample_list["token_type_ids"],
            sample_list["visual_embeddings"],
            sample_list["visual_embeddings_type"],
            getattr_torchscriptable(sample_list, "image_text_alignment", None),
            getattr_torchscriptable(sample_list, "masked_lm_labels", None),
        )

        return output_dict
```

```plain
        # 对状态字典进行格式和状态的规范化处理
        for key in list(copied_state_dict.keys()):
            formatted_key = cls.format_state_keycopy(key)
            copied_state_dict[formatted_key] = copied_state_dict.pop(key)

        # 加载模型, 并设置为推断模式
        instance.load_state_dict(copied_state_dict, strict=True)
        instance.eval()
        return instance

    def flatten(
        self,
        sample_list: Dict[str, Tensor],
        to_be_flattened: List[str],
        to_be_flattened_dim: List[str],
    ) -> Dict[str, Tensor]:
        for key in to_be_flattened:
            # 确保这些keys已经设置完毕, 否则将被置为None
            sample_list[key] = transform_to_batch_sequence(sample_list[key])

        for key in to_be_flattened_dim:
            sample_list[key] = transform_to_batch_sequence_dim(sample_list[key])

        return sample_list

    def add_post_flatten_params(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        sample_list["visual_embeddings_type"] = torch.zeros_like(sample_list["image_mask"])

        attention_mask = torch.cat((sample_list["input_mask"], sample_list["image_mask"]), dim=-1)

        sample_list["attention_mask"] = attention_mask

        if self.training_head_type == "pretraining":
            assert sample_list["masked_lm_labels"].size(-1) == sample_list["input_mask"].size(-1)
            new_lm_labels = torch.ones_like(attention_mask) * -1
            size_masked_lm_labels = sample_list["masked_lm_labels"].size()
            assert len(size_masked_lm_labels) == 2

            new_lm_labels[:size_masked_lm_labels[0], :size_masked_lm_labels[1]] = sample_list["masked_lm_labels"]
            sample_list["masked_lm_labels"] = new_lm_labels
        return sample_list

    def flatten_for_bert(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        to_be_flattened = ["input_ids", "token_type_ids", "input_mask", "image_mask"]

        to_be_flattened_dim = ["visual_embeddings"]

        if self.training_head_type == "pretraining":
            to_be_flattened.append("masked_lm_labels")
        # 把所有数据转化成形状等于 [batch, sequence_length, dim].
        flattened = self.flatten(sample_list, to_be_flattened, to_be_flattened_dim)
        return flattened

    def update_sample_list_based_on_head(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        bert_input_ids = sample_list["input_ids"]
        bert_input_mask = sample_list["input_mask"]
        bert_input_type_ids = sample_list["segment_ids"]
        if not torch.jit.is_scripting():
            image_info = getattr(sample_list, "image_info_0", {})
            image_dim_variable = getattr(image_info, "max_features", None)
            image_feat_variable = getattr(sample_list, "image_feature_0", None)
        else:
            image_feat_variable = sample_list["image_feature_0"]
            image_dim_variable = None

        if image_dim_variable is None:
            image_dim_variable = sample_list["image_feature_0"].new_full(
                                             size=(image_feat_variable.size(0), 1),
                                             fill_value=image_feat_variable.size(1),
                                             )
        # 更新样本值
        sample_list["visual_embeddings"] = image_feat_variable
        sample_list["image_dim"] = image_dim_variable
        sample_list["input_ids"] = bert_input_ids
        sample_list["input_mask"] = bert_input_mask
        sample_list["token_type_ids"] = bert_input_type_ids
        return sample_list

    def add_custom_params(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        visual_embeddings = sample_list["visual_embeddings"]
        image_dim = sample_list["image_dim"]
        # 如果是预训练模式, 则增加masked_lm_labels
        if self.training_head_type == "pretraining":
            sample_list["masked_lm_labels"] = sample_list["lm_label_ids"]
        # 添加image mask
        image_mask = torch.arange(
            visual_embeddings.size(-2), device=visual_embeddings.device
            ).expand(visual_embeddings.size()[:-1])

        # 如果图像的维度比image_mask的维度小则进行维度扩展
        if len(image_dim.size()) < len(image_mask.size()):
            image_dim = image_dim.unsqueeze(-1)
            assert len(image_dim.size()) == len(image_mask.size())

        image_mask = image_mask < image_dim
        sample_list["image_mask"] = image_mask.long()

        return sample_list

    # 为了实现对原始VisualBERT的后向兼容
    @classmethod
    def format_state_key(cls, key):
        return (
            key.replace("bert.bert", "model.bert")
            .replace("bert.cls", "model.cls")
            .replace("bert.classifier", "model.classifier")
        )

    # 获取优化器参数
    def get_optimizer_parameters(self, config):
        return get_optimizer_parameters_for_bert(self.model, config)

    def forward(self, sample_list: Dict[str, Tensor]) -> Dict[str, Tensor]:
        # pickle.dump(sample_list, open('sample_01.ckpt','wb'))
        # 更新sample_list名称为新的sample_list 名称及处理sample_list中的数据,这一步操作主要为预处理各种各样产生的字段名称不同，
        # 所以需要统一
        sample_list = self.update_sample_list_based_on_head(sample_list)
        # 为sample_list 增加自定义参数
        sample_list = self.add_custom_params(sample_list)
        # 将'input_ids', 'token_type_ids', 'input_mask', 'image_mask'转化为[batch, sequence_length, dim]
        sample_list = self.flatten_for_bert(sample_list)
        sample_list = self.add_post_flatten_params(sample_list)
        # pickle.dump(sample_list, open('sample_02.ckpt','wb'))

        # 依次传入BERT的参数, 得到返回值
        output_dict = self.model(
            sample_list["input_ids"],
            sample_list["input_mask"],
            sample_list["attention_mask"],
            sample_list["token_type_ids"],
            sample_list["visual_embeddings"],
            sample_list["visual_embeddings_type"],
            getattr_torchscriptable(sample_list, "image_text_alignment", None),
            getattr_torchscriptable(sample_list, "masked_lm_labels", None),
        )

        return output_dict
```
