+ 代码在文件sample.py中.

```plain
#SampleList 是针对Sample 进行批量处理
class SampleList(OrderedDict):
    _TENSOR_FIELD_ = "_tensor_field"
    def __init__(self, samples=None):
        super().__init__(self)
        if samples is None:
            samples = []
        # 如过长度为0则返回
        if len(samples) == 0:
            return
         # 加载dict 类型数据
        if self._check_and_load_dict(samples):
            return
        # 加载tuple 类型数据
        if self._check_and_load_tuple(samples):
            return
        # # 返回samples的所有key 
        fields = samples[0].keys()
        for field in fields:
            # samples 为空值的赋值
            if isinstance(samples[0][field], torch.Tensor):
                size = (len(samples), *samples[0][field].size())
                self[field] = samples[0][field].new_empty(size)
                if self._get_tensor_field() is None:
                    self._set_tensor_field(field)
            else:
            # 设置key 为空
                self[field] = [None for _ in range(len(samples))]
            for idx, sample in enumerate(samples):
            # 检查所有维度的size 是否相等
                if (
                    isinstance(sample[field], torch.Tensor)
                    and len(sample[field].size()) != 0
                    and sample[field].size(0) != samples[0][field].size(0)
                ):
                    raise AssertionError(
                        "Fields for all samples must be equally sized. "
                        "{} is of different sizes".format(field)
                    )
                # 对每一列的每一个位置进行填充
                self[field][idx] = self._get_data_copy(sample[field])
            if isinstance(samples[0][field], collections.abc.Mapping):
                self[field] = SampleList(self[field])
    # 检查并加载tuple 类型
    def _check_and_load_tuple(self, samples):
        if isinstance(samples[0], (tuple, list)) and isinstance(samples[0][0], str):
            for kv_pair in samples:
                self.add_field(kv_pair[0], kv_pair[1])
            return True
        else:
            return False
    # 检查并加载dict
    def _check_and_load_dict(self, samples):
        # 检查sample是是否为dict 类型
        if isinstance(samples, collections.abc.Mapping):
            # 遍历所有的samples，添加入fields 
            for key, value in samples.items():
                self.add_field(key, value)
            return True
        else:
            return False
    # 设置属性
    def __setattr__(self, key, value):
        self[key] = value
    # 返回属性
    def __getattr__(self, key):
        if key not in self:
            raise AttributeError(
                "Key {} not found in the SampleList. "
                "Valid choices are {}".format(key, self.fields())
            )
        fields = self.keys()
        if key in fields:
            return self[key]
        sample = Sample()
        for field in fields:
            sample[field] = self[field][key]
        return sample

    # 获取当前SampleList 的拷贝
    def copy(self):
        sample_list = SampleList()
        fields = self.fields()
        for field in fields:
            sample_list.add_field(field, self[field])
        return sample_list
    # 返回所有fields
    def fields(self):
        return list(self.keys())
    # 获取所有的fields 
    def get_fields(self, fields):
        current_fields = self.fields()
        return_list = SampleList()
        for field in fields:
            return_list.add_field(field, self[field])
        return return_list
    # 根据属性返回value
    def get_field(self, field):
        return self[field]
    # 返回数据
    def _get_data_copy(self, data):
        return data
    # 返回tensor 区域
    def _get_tensor_field(self):
        return self.__dict__.get(SampleList._TENSOR_FIELD_, None)
    # 设置tensor 区域
    def _set_tensor_field(self, value):
        self.__dict__[SampleList._TENSOR_FIELD_] = value
    # 返回一个SampleList的批量数据
    def get_batch_size(self):
        tensor_field = self._get_tensor_field()
        assert tensor_field is not None, "There is no tensor yet in SampleList"
        return self[tensor_field].size(0)
    # 向field 中添加data
    def add_field(self, field, data):
        # 遍历所有的fields
        fields = self.fields()
        # 默认fields 
        tensor_field = self._get_tensor_field()
        # 如果为dict类型
        if isinstance(data, collections.abc.Mapping):
            self[field] = SampleList(data)
        else:
            self[field] = self._get_data_copy(data)

            if isinstance(self[field], torch.Tensor) and tensor_field is None:
                self._set_tensor_field(field)
    # 与.to 函数类似，将SampleList的所有函数转化为一个特定的设备
    def to(self, device, non_blocking=True):
        # 获取所有keys
        fields = self.keys()
        # 深拷贝
        sample_list = self.copy()
        if not isinstance(device, torch.device):
            # 设备校验
            if not isinstance(device, str):
                raise TypeError(
                    "device must be either 'str' or "
                    "'torch.device' type, {} found".format(type(device))
                )
            device = torch.device(device)
                # 对于fields 中的每一个field调用.to方法
        for field in fields:
            if hasattr(sample_list[field], "to"):
                sample_list[field] = sample_list[field].to(
                    device, non_blocking=non_blocking
                )
                # 返回sample_list
        return sample_list
    # 锁页内存，当内存充足时设置为true，数据会直接从内存交换到显存
    def pin_memory(self):
        fields = self.keys()
        for field in fields:
            if hasattr(self[field], "pin_memory"):
                # 设置每一个区域锁页
                self[field] = self[field].pin_memory()
                # 返回self
        return self
```

```plain
def convert_batch_to_sample_list(
    batch: Union[SampleList, Dict[str, Any]]
) -> SampleList:
    # Create and return sample list with proper name
    # and type set if it is already not a sample list
    # (case of batched iterators)
    sample_list = batch
    if (
        # Check if batch is a list before checking batch[0]
        # or len as sometimes batch is already SampleList
        isinstance(batch, list)
        and len(batch) == 1
        and isinstance(batch[0], SampleList)
    ):
        sample_list = batch[0]
    elif not isinstance(batch, SampleList):
        sample_list = SampleList(batch)
    if sample_list._get_tensor_field() is None:
        sample_list = SampleList(sample_list.to_dict())
    return sample_list
```

```plain
import torch

class AnnotationDatabase(torch.utils.data.Dataset):
    # 处理标注数据的基类
    def __init__(self, config, path, *args, **kwargs):
        super().__init__()
        self.metadata = {}
        self.config = config
        self.start_idx = 0
        path = path
        self.load_annotation_db(path)

    def load_annotation_db(self, path):
        # 根据不同的后缀进行处理
        if path.endswith(".jsonl"):
            self._load_jsonl(path)
        elif path.endswith(".json"):
            self._load_json(path)
        else:
            raise ValueError("Unknown file format for annotation db")

    # 加载jsonl 格式数据
    def _load_jsonl(self, path):
        with PathManager.open(path, "r") as f:
            db = f.readlines()
            for idx, line in enumerate(db):
                db[idx] = json.loads(line.strip("\n"))
            self.data = db
            self.start_idx = 0

    # 加载json格式数据
    def _load_json(self, path):
        with PathManager.open(path, "r") as f:
            data = json.load(f)
        self.metadata = data.get("metadata", {})
        self.data = data.get("data", [])
        if len(self.data) == 0:
            raise RuntimeError("Dataset is empty")

    def __len__(self):
        return len(self.data) - self.start_idx

    def __getitem__(self, idx):
        data = self.data[idx + self.start_idx]
        return data

    def get_version(self):
        return self.metadata.get("version", None)
```

```plain
# 从.npy 和 _info.npy中读取数据
class ImageFeatureReader:
    def __init__(self, max_loc):
        self.max_loc = max_loc
        self.first = True
        self.take_item = False
    # 基于image_feat_path 读取数据
    def _load(self, image_feat_path):
        image_info = {}
        # 读入npy 文件的数据
        image_info["features"] = load_feat(image_feat_path)
        # 读入_info.npy 格式数据
        info_path = "{}_info.npy".format(image_feat_path.split(".npy")[0])
        # 更新image_info 读入_info 格式数据
        if PathManager.exists(info_path):
            image_info.update(load_feat(info_path).item())
        return image_info

    def read(self, image_feat_path):
        # 读取image_info 的信息
        image_info = self._load(image_feat_path)
        # 判断是否第一次读取
        if self.first:
            self.first = False
            if (
                image_info["features"].size == 1
                and "image_feat" in image_info["features"].item()
            ):
                self.take_item = True
        image_feature = image_info["features"]

        if self.take_item:
            item = image_info["features"].item()
            if "info" in item:
                if "image_text" in item["info"]:
                    image_info.update(item["info"])
                image_feature = item["feature"]

        # 读取image 的位置信息和 image维度信息
        image_loc, image_dim = image_feature.shape
        # 以下处理为提取到max_loc 位置的所有数据
        tmp_image_feat = np.zeros((self.max_loc, image_dim), dtype=np.float32)
        tmp_image_feat[0:image_loc, ] = image_feature[: self.max_loc, :]  # noqa
        image_feature = torch.from_numpy(tmp_image_feat)

        del image_info["features"]
        image_info["max_features"] = torch.tensor(image_loc, dtype=torch.long)
        return image_feature, image_info
```

```plain
# 构建图像特征数据库，将.npy 和_info.npy文件读入图像特征数据库
class FeaturesDatabase():
    def __init__(
        self, config, path, annotation_db=None, feature_key=None, *args, **kwargs
    ):

        self.feature_readers = []
        self.feature_dict = {}
        self.feature_key = config.get("feature_key", "feature_path")
        self.feature_key = feature_key if feature_key else self.feature_key
        path = path.split(",")
        # 初始化FeatureReader 对象
        for image_feature_dir in path:
            feature_reader = FeatureReader(
                base_path=image_feature_dir,
                depth_first=config.get("depth_first", False),
                max_features=config.get("max_features", 100),
            )
            self.feature_readers.append(feature_reader)
        self.paths = path
        self.annotation_db = annotation_db
        self._should_return_info = config.get("return_features_info", True)

    def _read_features_and_info(self, feat_file):
        features = []
        infos = []
        for feature_reader in self.feature_readers:
            feature, info = feature_reader.read(feat_file)
            features.append(feature)
            infos.append(info)
        if not self._should_return_info:
            infos = None
        return features, infos

    def _get_image_features_and_info(self, feat_file):
        assert isinstance(feat_file, str)
        image_feats, infos = self.feature_dict.get(feat_file, (None, None))
        if image_feats is None:
            image_feats, infos = self._read_features_and_info(feat_file)
        return image_feats, infos

    def __len__(self):
        # self._check_annotation_db_present()
        return len(self.annotation_db)

    def __getitem__(self, idx):
        # self._check_annotation_db_present()
        image_info = self.annotation_db[idx]
        return self.get(image_info)

    def get(self, item):
        feature_path = item.get(self.feature_key, None)
        if feature_path is None:
            feature_path = self._get_feature_path_based_on_image(item)
        return self.from_path(feature_path)

    def from_path(self, path):
        assert isinstance(path, str)
        if "genome" in path and path.endswith(".npy"):
            path = str(int(path.split("_")[-1].split(".")[0])) + ".npy"
        features, infos = self._get_image_features_and_info(path)
        item = {}
        for idx, image_feature in enumerate(features):
            item["image_feature_%s" % idx] = image_feature
            if infos is not None:
                # infos[idx].pop("cls_prob", None)
                item["image_info_%s" % idx] = infos[idx]
        return item

    def _get_feature_path_based_on_image(self, item):
        image_path = self._get_attrs(item)[0]
        if isinstance(image_path, int):
            return f"{image_path}.npy"
        feature_path = ".".join(image_path.split(".")[:-1]) + ".npy"
        return feature_path
```
