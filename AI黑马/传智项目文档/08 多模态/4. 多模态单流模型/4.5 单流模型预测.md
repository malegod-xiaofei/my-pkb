```plain
class MultiTrainer(pl.LightningModule):
    def __init__(self, config, type='hm'):
        super().__init__()
        self.save_hyperparameters()
        self.config = config
        self.training_head_type = self.config.training_head_type
        if self.config.use_pretrain:
            cls = str_to_class(self.config.modelname)
            self.model = cls.from_pretrained(self.config.pretrain_checkpoint , self.config)
        else:
            cls = str_to_class(self.config.modelname)
            self.model = cls(self.config)


    @classmethod
    def format_state_key(cls, key):
        return (
            key.replace("bert.bert", "model.bert")
            .replace("bert.cls", "model.cls")
            .replace("bert.classifier", "model.classifier"))

    @staticmethod
    def add_model_specific_args(parent_parser):
        parser = ArgumentParser(parents=[parent_parser], add_help=False)
        parser.add_argument('--learning_rate', type=float, default=0.0001)
        parser.add_argument('--model_type', type=str, default="hm")
        parser.add_argument('--config_path', type=str,
                            default="./defaults.yaml")
        parser.add_argument('--gpus', type=str,
                            default="3")
        return parser

    def get_optimizer_parameters(self, config):
        return get_optimizer_parameters_for_bert(self.model, config)

    def input_samplelist_build_(self, batch):
        # construct the sample list based one the batch input
        input_ids = batch['input_ids']
        input_mask = batch['input_mask']  # [0] * len(input_ids)
        segment_ids = batch['segment_ids']
        visual_embeddings = batch['image_feature_0']
        targets = batch['targets']
        image_info_0 = batch['image_info_0']

        sample_list = SampleList()
        sample_list.add_field(
            "input_ids",
            input_ids
        )
        sample_list.add_field("input_mask", input_mask)
        sample_list.add_field("segment_ids", segment_ids)
        sample_list.add_field("image_feature_0", visual_embeddings)
        sample_list.add_field(
            "lm_label_ids", batch['lm_label_ids']
        )
        sample_list.add_field('image_info_0', image_info_0)
        sample_list.dataset_name = "hm"
        sample_list.dataset_type = "hate-speech"
        sample_list = sample_list.to(get_current_device())

        return sample_list, targets

    def training_step(self, batch, batch_idx):
        # construct the input samplelist build
        sample_list, targets = self.input_samplelist_build_(batch)
        # sample_list to device
        sample_list = sample_list.to(get_current_device())
        y_hat = self.model(sample_list)
        y = targets
        y_hat = y_hat['scores']
        loss = F.cross_entropy(y_hat, y)
        preds = y_hat.detach().cpu().numpy()
        out_label_ids = y.detach().cpu().numpy()
        # add log to the tensorboard
        self.log('train_loss', loss, on_step=True,
                 on_epoch=True, prog_bar=True, logger=True)
        outputs = {'loss': loss, 'pred': preds,
                   "target": out_label_ids,
                   }
        return outputs

    def _metrics_output(self, outputs) -> tuple:

        val_loss_mean = torch.stack([x["val_loss"] for x in outputs
                                     ]).mean().detach().cpu().item()
        preds = np.concatenate([x["pred"] for x in outputs], axis=0)
        output_softmax = np.concatenate(
            [x["output_softmax"] for x in outputs], axis=0)
        preds = np.argmax(preds, axis=1)
        out_label_ids = np.concatenate([x["target"] for x in outputs], axis=0)
        acc_ = accuracy_score(preds, out_label_ids)
        targets = np.concatenate([x["targets"] for x in outputs], axis=0)
        avg_percision = average_precision_score(targets, output_softmax)
        try:
            roc_auc = roc_auc_score(targets, output_softmax)
        except:
            roc_auc = 0
        return acc_, roc_auc, avg_percision

    def calculate_loss_val_test(self, batch):
        # print(batch)
        sample_list, targets = self.input_samplelist_build_(batch)
        sample_list = sample_list.to(get_current_device())
        y_hat = self.model(sample_list)
        y = targets
        y_hat = y_hat['scores']
        output_softmax = F.softmax(y_hat, dim=-1)
        y = y.to(get_current_device())
        loss = F.cross_entropy(y_hat, y)
        tmp_eval_loss, logits = loss, y_hat
        preds = logits.detach().cpu().numpy()
        out_predict = np.argmax(preds, axis=1)
        target_y = y.detach().cpu().numpy()
        target_convert_onehot = _convert_to_one_hot(targets, output_softmax)

        return tmp_eval_loss.detach().cpu(), preds, target_y, output_softmax.cpu(), target_convert_onehot.cpu()

    def validation_step(self, batch, batch_idx):

        loss, preds, target_y, output_softmax, target_convert_onehot = self.calculate_loss_val_test(
            batch)
        self.log('val_loss', loss, on_step=True,
                 on_epoch=True, prog_bar=True, logger=True)
        return {
            "val_loss": loss,
            "pred": preds,
            "target": target_y,
            "output_softmax":  output_softmax,
            "targets": target_convert_onehot
        }

    def validation_epoch_end(self, outputs) -> dict:

        acc_, roc_auc, avg_percion = self._metrics_output(outputs)
        print(acc_, roc_auc, avg_percion)
        self.log('acc', acc_,
                 on_epoch=True, prog_bar=True, logger=True)
        self.log('roc_auc', roc_auc,
                 on_epoch=True, prog_bar=True, logger=True)
        self.log('avg_percion', avg_percion,
                 on_epoch=True, prog_bar=True, logger=True)
        return {
            "acc": acc_,
            "roc_auc": roc_auc,
            "avg_percion": avg_percion
        }

    def test_step(self, batch, batch_idx):

        loss, preds, target_y, output_softmax, target_convert_onehot = self.calculate_loss_val_test(
            batch)

        self.log('test_loss', loss, on_step=True,
                 on_epoch=True, prog_bar=True, logger=True)

        return {
            "val_loss": loss,
            "pred": preds,
            "target": target_y,
            "output_softmax":  output_softmax,
            "targets": target_convert_onehot
        }

    def test_epoch_end(self, outputs) -> dict:
        acc_, roc_auc, avg_percion = self._metrics_output(outputs)

        return {
            "acc": acc_,
            "roc_auc": roc_auc,
            "avg_percion": avg_percion
        }

    def configure_optimizers(self):
        optimizer = build_optimizer(self, self.config)
        num_training_steps = self.config.scheduler.params.num_training_steps #22000
        num_warmup_steps =  self.config.scheduler.params.num_warmup_steps #2000
        lr_scheduler = get_linear_schedule_with_warmup(
            optimizer, num_warmup_steps,num_training_steps, last_epoch=-1)
        return {
            "optimizer": optimizer,
            "lr_scheduler": {"scheduler": lr_scheduler, "interval": "step"},
        }

    @staticmethod
    def add_model_specific_args(parent_parser):
        parser = ArgumentParser(parents=[parent_parser], add_help=False)
        parser.add_argument('--model_type', type=str, default="hm")
        parser.add_argument('--config_path', type=str,
                            default="./defaults.yaml")
        return parser

def cli_main():
    pl.seed_everything(1234)
    parser = ArgumentParser()
    parser = pl.Trainer.add_argparse_args(parser)
    parser = MultiTrainer.add_model_specific_args(parser)
    parser = hatespeech_datamodule.add_argparse_args(parser)
    args = parser.parse_args()

    config = load_yaml(args.config_path)

    dm = hatespeech_datamodule(config.dataset)
    tblogger = TensorBoardLogger('./tb_logs',name=str(config.tag_name) )
    model = MultiTrainer(config)
    model = model.to(get_current_device())
    trainer = pl.Trainer(
        # accelerator="dp",
        auto_scale_batch_size=True,
        logger=tblogger, 
        val_check_interval=0.1,
        log_every_n_steps=3,
        sync_batchnorm=True,
        num_sanity_val_steps=-1,
        replace_sampler_ddp=False,
        ).from_argparse_args(args)

    model_save = './deploy/modelvisual_bert_nopretrain.ckpt'
    result = trainer.test(model, datamodule=dm, ckpt_path=model_save)
    print(result)    
if __name__ == '__main__':
    cli_main()
```
