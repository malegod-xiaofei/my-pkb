```plain
# 多模态训练主类
class MultiTrainer(pl.LightningModule):
    def __init__(self, config, type='hm'):
        super().__init__()
        # 明确若干关键参数
        self.save_hyperparameters()
        self.config = config
        # 明确当前训练的任务类型头
        self.training_head_type = self.config.training_head_type
        # 使用预训练模型, 则加载
        if self.config.use_pretrain:
            cls = str_to_class(self.config.modelname)
            self.model = cls.from_pretrained(self.config.pretrain_checkpoint , self.config)
        # 否则, 初始化模型
        else:
            cls = str_to_class(self.config.modelname)
            self.model = cls(self.config)


    # 该方法为了模型的后向兼容
    @classmethod
    def format_state_key(cls, key):
        return (key.replace("bert.bert", "model.bert")
                   .replace("bert.cls", "model.cls")
                   .replace("bert.classifier", "model.classifier"))

    # 添加模型训练的命令行可选参数
    @staticmethod
    def add_model_specific_args(parent_parser):
        parser = ArgumentParser(parents=[parent_parser], add_help=False)
        parser.add_argument('--learning_rate', type=float, default=0.0001)
        parser.add_argument('--model_type', type=str, default="hm")
        parser.add_argument('--config_path', type=str, default="./defaults.yaml")
        parser.add_argument('--gpus', type=str, default="3")
        return parser


    # 获取优化器参数的函数
    def get_optimizer_parameters(self, config):
        return get_optimizer_parameters_for_bert(self.model, config)


    # 构建输入样本列表
    def input_samplelist_build_(self, batch):
        input_ids = batch['input_ids']
        # mask张量的形状: [0] * len(input_ids)
        input_mask = batch['input_mask']
        segment_ids = batch['segment_ids']
        visual_embeddings = batch['image_feature_0']
        targets = batch['targets']
        image_info_0 = batch['image_info_0']

        # 初始化样本列表对象
        sample_list = SampleList()
        # 添加对象域和具体的值
        sample_list.add_field("input_ids", input_ids)
        sample_list.add_field("input_mask", input_mask)
        sample_list.add_field("segment_ids", segment_ids)
        sample_list.add_field("image_feature_0", visual_embeddings)
        sample_list.add_field("lm_label_ids", batch['lm_label_ids'])
        sample_list.add_field('image_info_0', image_info_0)
        sample_list.dataset_name = "hm"
        sample_list.dataset_type = "hate-speech"
        # 最后将组装好的张量列表数据发送到训练设备上(GPU or CPU)
        sample_list = sample_list.to(get_current_device())

        return sample_list, targets


    # 一个batch_size的训练
    def training_step(self, batch, batch_idx):
        # 第1步: 创建出当前batch的训练数据
        sample_list, targets = self.input_samplelist_build_(batch)

        # 第2步: 将训练数据发送到训练设备上(GPU or CPU)
        sample_list = sample_list.to(get_current_device())

        # 第3步: 输入模型进行计算
        y_hat = self.model(sample_list)

        # 第4步: 提取模型的预测y_hat, 以及真实标签y
        targets = targets.to(get_current_device())
        sample_list = sample_list.to(get_current_device())
        y = targets
        y_hat = y_hat['scores']

        # 第5步: 计算交叉熵损失值
        loss = F.cross_entropy(y_hat, y)

        # 第6步: 转换成numpy类型数据返回
        preds = y_hat.detach().cpu().numpy()
        out_label_ids = y.detach().cpu().numpy()

        # 第7步: 向Tensorboard中添加训练日志信息
        self.log('train_loss', loss, on_step=True, on_epoch=True, prog_bar=True, logger=True)

        # 第8步: 以字典类型组装返回张量
        outputs = {'loss': loss, 'pred': preds, "target": out_label_ids}

        return outputs


    # 对输出张量结果计算关键指标的函数
    def _metrics_output(self, outputs) -> tuple:
        # 第1步: 计算验证集上的平均损失值
        val_loss_mean = torch.stack([x["val_loss"] for x in outputs]).mean().detach().cpu().item()

        # 第2步: 将输出字典中的预测张量叠加起来
        preds = np.concatenate([x["pred"] for x in outputs], axis=0)

        # 第3步: 将输出字典中的softmax分布张量叠加起来
        output_softmax = np.concatenate([x["output_softmax"] for x in outputs], axis=0)

        # 第4步: 在preds第1个维度上计算最大值
        preds = np.argmax(preds, axis=1)

        # 第5步: 将输出字典中的标签值叠加起来
        out_label_ids = np.concatenate([x["target"] for x in outputs], axis=0)

        # 第6步: 利用预测值preds, 和标签值out_label_ids, 进行准确率的计算
        acc_ = accuracy_score(preds, out_label_ids)

        # 第7步: 将输出字典中的标签值叠加起来
        targets = np.concatenate([x["targets"] for x in outputs], axis=0)

        # 第8步: 计算平均精确率的值
        avg_percision = average_precision_score(targets, output_softmax)

        # 第9步: 如果可以, 计算ROC分值, 否则置为0
        try:
            roc_auc = roc_auc_score(targets, output_softmax)
        except:
            roc_auc = 0

        return acc_, roc_auc, avg_percision


    # 计算验证集, 测试集上的损失值和预测值的函数
    def calculate_loss_val_test(self, batch):
        # 构建验证集, 测试集的样本列表和标签列表
        sample_list, targets = self.input_samplelist_build_(batch)
        sample_list = sample_list.to(get_current_device())

        # 经历模型的预测, 得到输出张量结果
        y_hat = self.model(sample_list)
        targets = targets.to(get_current_device())
        sample_list = sample_list.to(get_current_device())
        y = targets
        y_hat = y_hat['scores']

        # 利用模型的预测值和真实标签值, 计算交叉熵损失值
        output_softmax = F.softmax(y_hat, dim=-1)
        loss = F.cross_entropy(y_hat, y)

        # 处理若干关键值并返回
        tmp_eval_loss, logits = loss, y_hat
        preds = logits.detach().cpu().numpy()
        out_predict = np.argmax(preds, axis=1)
        target_y = y.detach().cpu().numpy()
        target_convert_onehot = _convert_to_one_hot(targets, output_softmax)

        return tmp_eval_loss.detach().cpu(), preds, target_y, output_softmax.cpu(), target_convert_onehot.cpu()


    # 对一个batch的验证集进行计算关键指标并返回结果的函数
    def validation_step(self, batch, batch_idx):
        # 计算损失值和预测值
        loss, preds, target_y, output_softmax, target_convert_onehot = self.calculate_loss_val_test(batch)

        # 向Tensorboard中添加日志信息
        self.log('val_loss', loss, on_step=True, on_epoch=True, prog_bar=True, logger=True)

        # 以字典类型返回信息
        return {"val_loss": loss,
                "pred": preds,
                "target": target_y,
                "output_softmax":  output_softmax,
                "targets": target_convert_onehot
                }


    # 一个epoch训练结束后, 进行依次验证集上的评估
    def validation_epoch_end(self, outputs) -> dict:
        # 针对输出结果张量, 计算关键指标
        acc_, roc_auc, avg_percion = self._metrics_output(outputs)
        print(acc_, roc_auc, avg_percion)

        # 添加日志信息, 三个关键指标
        self.log('acc', acc_, on_epoch=True, prog_bar=True, logger=True)
        self.log('roc_auc', roc_auc, on_epoch=True, prog_bar=True, logger=True)
        self.log('avg_percion', avg_percion, on_epoch=True, prog_bar=True, logger=True)

        # 以字典类型返回关键指标信息
        return {"acc": acc_,
                "roc_auc": roc_auc,
                "avg_percion": avg_percion
                }


    # 一个batch的样本数据进行测试的函数
    def test_step(self, batch, batch_idx):
        # 对于一个batch的测试集数据直接计算损失值和预测值
        loss, preds, target_y, output_softmax, target_convert_onehot = self.calculate_loss_val_test(batch)
        print("output >>>>", loss, preds )

        # 添加日志信息
        self.log('test_loss', loss, on_step=True, on_epoch=True, prog_bar=True, logger=True)

        # 以字典类型返回信息
        return {"val_loss": loss,
                "pred": preds,
                "target": target_y,
                "output_softmax":  output_softmax,
                "targets": target_convert_onehot
                }


    # 一个epoch训练结束后, 进行一次测试
    def test_epoch_end(self, outputs) -> dict:
        acc_, roc_auc, avg_percion = self._metrics_output(outputs)

        return {"acc": acc_,
                "roc_auc": roc_auc,
                "avg_percion": avg_percion
               }

    # 设置优化器的关键参数
    def configure_optimizers(self):
        # 创建优化器对象
        optimizer = build_optimizer(self, self.config)

        # 设置训练步数, 默认为22000
        num_training_steps = self.config.scheduler.params.num_training_steps

        # 设置warmup的步数, 默认为2000
        num_warmup_steps =  self.config.scheduler.params.num_warmup_steps

        # 实例化优化器的调度器
        lr_scheduler = get_linear_schedule_with_warmup(optimizer, num_warmup_steps, num_training_steps, last_epoch=-1)

        # 以字典类型返回优化器和调度器
        return {"optimizer": optimizer,
                "lr_scheduler": {"scheduler": lr_scheduler, "interval": "step"},
               }

    # 添加模型训练的命令行可选参数
    @staticmethod
    def add_model_specific_args(parent_parser):
        parser = ArgumentParser(parents=[parent_parser], add_help=False)
        parser.add_argument('--model_type', type=str, default="hm")
        parser.add_argument('--config_path', type=str, default="./defaults.yaml")
        return parser
```

```plain
# 命令行模式的主函数定义
def cli_main():
    # 实例化Parser对象
    parser = ArgumentParser()
    parser = pl.Trainer.add_argparse_args(parser)
    # 向MultiTrainer和hatespeech_datamodule中添加参数信息
    parser = MultiTrainer.add_model_specific_args(parser)
    parser = hatespeech_datamodule.add_argparse_args(parser)
    args = parser.parse_args()

    # 导入yaml文件中的配置信息
    config = load_yaml(args.config_path)

    # 第1步: 导入数据集
    dm = hatespeech_datamodule(config.dataset)

    # 第2步: 实例化Tensorboard日志信息对象
    tblogger = TensorBoardLogger('./tb_logs',name=str(config.tag_name) )

    # 第3步: 实例化MultiTrainer模型
    model = MultiTrainer(config)
    model = model.to(get_current_device())

    # 第4步: 实例化真实的训练对象
    # check_val_every_n_epoch=1 : 每训练1个epoch校验1次, 即自动调用validation_step()函数.
    # val_check_interval=0.25 : 每训练单个epoch的25%调用校验函数一次, 注意需要传入float数值.
    # val_check_interval=100 : 单个epoch内, 每训练完100个batch数据调用校验函数一次, 注意传入int数值.
    # limit_test_batches=0.05 : 取测试集中的5%的数据来进行测试.
    # num_sanity_val_steps=0, 2, -1 : 在正式开启训练模型之前, 是否做若干轮的validation测试.
    #                                 0 表示不做; 
    #                                -1 表示做完整个validation epoch才开启训练;
    #                                 2 表示正式开启训练前先做2个batch的validation.
    # replace_sampler_ddp=False : 多GPU训练时, 需要为DataLoader指定Sample, 来为GPU分配数据.
    #                             Ligtning默认进行此操作, 对训练集shuffle, 对测试集不做shuffle.
    #                             设置为False, 则指定自定义的Sampler
    # sync_batchnorm=True : Batch Normalizetion对模型的训练影响很大, 尤其是CV领域, 所以默认会同步所有GPU上的BN.
    # logger : 指定日志记录对象.
    # log_every_n_steps : 每训练n个step记录一次日志信息.
    # progress_bar_refresh_rate=1 : 每个step更新一次进度条, 设置为0则代表关闭进度条.
    trainer = pl.Trainer(num_sanity_val_steps=0,
                         logger=tblogger,
                         val_check_interval=0.1,
                         log_every_n_steps=3,
                         sync_batchnorm=True,
                         replace_sampler_ddp=False,
                        ).from_argparse_args(args)

    # 第5步: 模型训练
    trainer.fit(model, datamodule=dm)

    # 第6步: 模型保存
    model_save = './model' + str(config.tag_name) +".ckpt"
    # dm = dm.to(get_current_device())
    # trainer.saving
    # trainer.save_checkpoint(model_save)    

    # 第7步: 模型的测试
    result = trainer.test(model, datamodule=dm, ckpt_path=model_save)
```
